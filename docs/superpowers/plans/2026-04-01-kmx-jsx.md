# KMX (JSX Support) Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add JSX syntax to KimchiLang via `.kmx` files with a compiler plugin system, compiling to React 19's jsx-runtime.

**Architecture:** A plugin system with hook points in lexer/parser/generator/typechecker. The KMX-React plugin implements JSX lexing (tag detection, attribute parsing), JSX parsing (elements, fragments, expressions, text), and JSX generation (jsx/jsxs calls with children handling, key extraction, auto-import). A stdlib module extern's the full React 19 API.

**Tech Stack:** Pure JavaScript, zero dependencies. React 19 via extern declarations.

```javascript
  '.kmx': './kmx-react.js',
};

const loadedPlugins = new Map();

export async function loadPluginsForFile(filePath) {
  if (!filePath) return [];
  
  const ext = filePath.match(/\.\w+$/)?.[0];
  if (!ext || !EXTENSION_PLUGINS[ext]) return [];
  
  const pluginPath = EXTENSION_PLUGINS[ext];
  
  if (!loadedPlugins.has(pluginPath)) {
    try {
      const mod = await import(pluginPath);
      loadedPlugins.set(pluginPath, mod.default);
    } catch (e) {
      throw new Error(`Failed to load plugin for ${ext}: ${e.message}`);
    }
  }
  
  return [loadedPlugins.get(pluginPath)];
}

export function loadPluginsForFileSync(filePath) {
  // For synchronous contexts, return empty — plugins loaded via compile options
  return [];
}
```

Create `src/extensions/kmx-react.js`:

```javascript
// KMX-React Compiler Extension
// Adds JSX syntax support for .kmx files, compiling to React 19 jsx-runtime

const JSX_TOKEN_TYPES = {
  JSX_OPEN: 'JSX_OPEN',         // <tag
  JSX_CLOSE: 'JSX_CLOSE',       // </tag>
  JSX_SELF_CLOSE: 'JSX_SELF_CLOSE', // />
  JSX_TEXT: 'JSX_TEXT',          // raw text between tags
  JSX_EXPR_START: 'JSX_EXPR_START', // { in JSX context
  JSX_EXPR_END: 'JSX_EXPR_END',    // } in JSX context
  JSX_GT: 'JSX_GT',             // > closing opening tag
  JSX_FRAGMENT_OPEN: 'JSX_FRAGMENT_OPEN',   // <>
  JSX_FRAGMENT_CLOSE: 'JSX_FRAGMENT_CLOSE', // </>
};

function lexerRules(lexer) {
  // Only activate in expression context when we see <
  // Need to distinguish < (less-than) from <tag (JSX)
  if (lexer.peek() !== '<') return null;
  
  const saved = lexer.pos;
  const savedLine = lexer.line;
  const savedCol = lexer.column;
  
  lexer.advance(); // consume <
  
  // Check what follows <
  if (lexer.peek() === '/') {
    // Closing tag: </tag>
    lexer.advance(); // consume /
    if (lexer.peek() === '>') {
      // Fragment close: </>
      lexer.advance(); // consume >
      return { type: JSX_TOKEN_TYPES.JSX_FRAGMENT_CLOSE, value: '', line: savedLine, column: savedCol };
    }
    let tagName = '';
    while (/[a-zA-Z0-9_.]/.test(lexer.peek())) {
      tagName += lexer.advance();
    }
    if (tagName && lexer.peek() === '>') {
      lexer.advance(); // consume >
      return { type: JSX_TOKEN_TYPES.JSX_CLOSE, value: tagName, line: savedLine, column: savedCol };
    }
    // Not a closing tag — backtrack
    lexer.pos = saved;
    lexer.line = savedLine;
    lexer.column = savedCol;
    return null;
  }
  
  if (lexer.peek() === '>') {
    // Fragment open: <>
    lexer.advance(); // consume >
    return { type: JSX_TOKEN_TYPES.JSX_FRAGMENT_OPEN, value: '', line: savedLine, column: savedCol };
  }
  
  // Opening tag: <tagName
  if (/[a-zA-Z_]/.test(lexer.peek())) {
    let tagName = '';
    while (/[a-zA-Z0-9_.]/.test(lexer.peek())) {
      tagName += lexer.advance();
    }
    
    // This is a JSX opening tag
    // We emit JSX_OPEN, then let the normal lexer handle attributes
    // The parser will know to expect JSX attributes until > or />
    
    // But we need to handle attributes ourselves since they use JSX rules
    // (e.g., {expr} for dynamic values, = for assignment)
    const attrs = [];
    
    // Skip whitespace and lex attributes until > or />
    while (lexer.peek() !== '>' && !(lexer.peek() === '/' && lexer.source[lexer.pos + 1] === '>') && lexer.peek() !== '\0') {
      // Skip whitespace
      while (/\s/.test(lexer.peek())) lexer.advance();
      
      if (lexer.peek() === '>' || (lexer.peek() === '/' && lexer.source[lexer.pos + 1] === '>')) break;
      
      // Read attribute name
      let attrName = '';
      while (/[a-zA-Z0-9_\-]/.test(lexer.peek())) {
        attrName += lexer.advance();
      }
      
      if (!attrName) { lexer.advance(); continue; } // skip unexpected char
      
      let attrValue = null;
      
      // Check for =
      while (/\s/.test(lexer.peek())) lexer.advance();
      if (lexer.peek() === '=') {
        lexer.advance(); // consume =
        while (/\s/.test(lexer.peek())) lexer.advance();
        
        if (lexer.peek() === '"' || lexer.peek() === "'") {
          // String value
          const quote = lexer.advance();
          let val = '';
          while (lexer.peek() !== quote && lexer.peek() !== '\0') {
            val += lexer.advance();
          }
          if (lexer.peek() === quote) lexer.advance();
          attrValue = { type: 'string', value: val };
        } else if (lexer.peek() === '{') {
          // Expression value — collect until matching }
          lexer.advance(); // consume {
          let depth = 1;
          let expr = '';
          while (depth > 0 && lexer.peek() !== '\0') {
            if (lexer.peek() === '{') depth++;
            else if (lexer.peek() === '}') { depth--; if (depth === 0) break; }
            expr += lexer.advance();
          }
          if (lexer.peek() === '}') lexer.advance();
          attrValue = { type: 'expression', value: expr.trim() };
        }
      } else {
        // Boolean attribute (no value) — like `disabled`
        attrValue = { type: 'boolean', value: true };
      }
      
      attrs.push({ name: attrName, value: attrValue });
    }
    
    // Check for self-closing or regular close
    let selfClosing = false;
    if (lexer.peek() === '/' && lexer.source[lexer.pos + 1] === '>') {
      lexer.advance(); // /
      lexer.advance(); // >
      selfClosing = true;
    } else if (lexer.peek() === '>') {
      lexer.advance(); // >
    }
    
    // Now lex children if not self-closing
    const children = [];
    if (!selfClosing) {
      children.push(...lexJSXChildren(lexer, tagName));
    }
    
    // Return a compound token that the parser can consume directly
    return {
      type: JSX_TOKEN_TYPES.JSX_OPEN,
      value: tagName,
      attrs,
      children,
      selfClosing,
      line: savedLine,
      column: savedCol,
    };
  }
  
  // Not JSX — backtrack
  lexer.pos = saved;
  lexer.line = savedLine;
  lexer.column = savedCol;
  return null;
}

function lexJSXChildren(lexer, parentTag) {
  const children = [];
  
  while (lexer.peek() !== '\0') {
    // Check for closing tag
    if (lexer.peek() === '<' && lexer.source[lexer.pos + 1] === '/') {
      // Verify it's our closing tag
      const saved = lexer.pos;
      lexer.advance(); // <
      lexer.advance(); // /
      let closeName = '';
      while (/[a-zA-Z0-9_.]/.test(lexer.peek())) {
        closeName += lexer.advance();
      }
      if (closeName === parentTag && lexer.peek() === '>') {
        lexer.advance(); // >
        break; // end of this element
      }
      // Not our closing tag — backtrack
      lexer.pos = saved;
    }
    
    // Check for child element or fragment
    if (lexer.peek() === '<') {
      const childToken = lexerRules(lexer);
      if (childToken) {
        children.push(childToken);
        continue;
      }
    }
    
    // Check for expression {expr}
    if (lexer.peek() === '{') {
      lexer.advance(); // consume {
      let depth = 1;
      let expr = '';
      while (depth > 0 && lexer.peek() !== '\0') {
        if (lexer.peek() === '{') depth++;
        else if (lexer.peek() === '}') { depth--; if (depth === 0) break; }
        expr += lexer.advance();
      }
      if (lexer.peek() === '}') lexer.advance();
      children.push({ type: 'JSX_EXPRESSION', value: expr.trim() });
      continue;
    }
    
    // Text content
    let text = '';
    while (lexer.peek() !== '<' && lexer.peek() !== '{' && lexer.peek() !== '\0') {
      text += lexer.advance();
    }
    if (text.trim()) {
      children.push({ type: JSX_TOKEN_TYPES.JSX_TEXT, value: text.trim() });
    }
  }
  
  return children;
}

// Parser rules — convert JSX tokens to AST nodes
function parserRules(parser) {
  // Check if current token is a JSX_OPEN
  if (parser.check && parser.peek() && parser.peek().type === JSX_TOKEN_TYPES.JSX_OPEN) {
    const token = parser.advance();
    return buildJSXNode(token, parser);
  }
  if (parser.check && parser.peek() && parser.peek().type === JSX_TOKEN_TYPES.JSX_FRAGMENT_OPEN) {
    parser.advance();
    // Fragment — find matching JSX_FRAGMENT_CLOSE
    // Children were already collected by the lexer
    // Actually fragments are handled differently — let me reconsider
    return { type: 'JSXFragment', children: [] };
  }
  return null;
}

function buildJSXNode(token, parser) {
  const tag = /^[A-Z]/.test(token.value) 
    ? { type: 'Identifier', name: token.value }
    : token.value;
  
  // Parse attribute expression values through the real parser
  const attributes = (token.attrs || []).map(attr => {
    let value;
    if (attr.value === null || attr.value === undefined) {
      value = { type: 'Literal', value: true };
    } else if (attr.value.type === 'string') {
      value = { type: 'Literal', value: attr.value.value, isString: true };
    } else if (attr.value.type === 'boolean') {
      value = { type: 'Literal', value: true };
    } else if (attr.value.type === 'expression') {
      // Parse the expression string through the KimchiLang parser
      const { tokenize } = parser.constructor._imports || {};
      // We need access to tokenize — store it on the parser or import it
      // For now, store expression as raw string — generator will handle
      value = { type: 'JSXRawExpression', source: attr.value.value };
    }
    return { name: attr.name, value };
  });
  
  // Convert children
  const children = (token.children || []).map(child => {
    if (child.type === JSX_TOKEN_TYPES.JSX_OPEN) {
      return buildJSXNode(child, parser);
    }
    if (child.type === JSX_TOKEN_TYPES.JSX_FRAGMENT_OPEN) {
      return { type: 'JSXFragment', children: (child.children || []).map(c => buildJSXNode(c, parser)) };
    }
    if (child.type === 'JSX_EXPRESSION') {
      return { type: 'JSXExpression', source: child.value };
    }
    if (child.type === JSX_TOKEN_TYPES.JSX_TEXT) {
      return { type: 'JSXText', value: child.value };
    }
    return child;
  });
  
  return {
    type: 'JSXElement',
    tag,
    attributes,
    children,
    selfClosing: token.selfClosing,
    line: token.line,
    column: token.column,
  };
}

// Generator visitors — emit jsx()/jsxs() calls
function generatorVisitors(generator, node) {
  if (node.type === 'JSXElement') return visitJSXElement(generator, node);
  if (node.type === 'JSXFragment') return visitJSXFragment(generator, node);
  if (node.type === 'JSXExpression') return visitJSXExpression(generator, node);
  if (node.type === 'JSXText') return JSON.stringify(node.value);
  if (node.type === 'JSXRawExpression') return node.source;
  return undefined;
}

function visitJSXElement(gen, node) {
  const tag = typeof node.tag === 'string' ? JSON.stringify(node.tag) : node.tag.name;
  
  // Build props object
  const props = {};
  let keyValue = null;
  
  for (const attr of node.attributes || []) {
    if (attr.name === 'key') {
      keyValue = attrValueToJS(gen, attr.value);
      continue;
    }
    props[attr.name] = attrValueToJS(gen, attr.value);
  }
  
  // Handle children
  const validChildren = (node.children || []).filter(c => c.type !== 'JSXText' || c.value.trim());
  
  if (validChildren.length === 1) {
    props.children = childToJS(gen, validChildren[0]);
  } else if (validChildren.length > 1) {
    props.children = `[${validChildren.map(c => childToJS(gen, c)).join(', ')}]`;
  }
  
  // Build props string
  const propsEntries = Object.entries(props);
  let propsStr;
  if (propsEntries.length === 0) {
    propsStr = '{}';
  } else {
    const parts = propsEntries.map(([k, v]) => {
      // Handle hyphenated attribute names
      const key = k.includes('-') ? JSON.stringify(k) : k;
      return `${key}: ${v}`;
    });
    propsStr = `{ ${parts.join(', ')} }`;
  }
  
  // Choose jsx vs jsxs
  const fn = validChildren.length > 1 ? 'jsxs' : 'jsx';
  
  if (keyValue) {
    return `${fn}(${tag}, ${propsStr}, ${keyValue})`;
  }
  return `${fn}(${tag}, ${propsStr})`;
}

function visitJSXFragment(gen, node) {
  const children = (node.children || []).filter(c => c.type !== 'JSXText' || c.value.trim());
  
  if (children.length === 0) {
    return 'jsx(Fragment, {})';
  }
  if (children.length === 1) {
    return `jsx(Fragment, { children: ${childToJS(gen, children[0])} })`;
  }
  return `jsxs(Fragment, { children: [${children.map(c => childToJS(gen, c)).join(', ')}] })`;
}

function visitJSXExpression(gen, node) {
  if (node.source) return node.source;
  if (node.expression) return gen.visitExpression(node.expression);
  return 'null';
}

function attrValueToJS(gen, value) {
  if (!value) return 'true';
  if (value.type === 'Literal') {
    if (value.isString) return JSON.stringify(value.value);
    return String(value.value);
  }
  if (value.type === 'JSXRawExpression') return value.source;
  return gen.visitExpression(value);
}

function childToJS(gen, child) {
  if (child.type === 'JSXText') return JSON.stringify(child.value);
  if (child.type === 'JSXExpression') return child.source || gen.visitExpression(child.expression);
  if (child.type === 'JSXElement') return visitJSXElement(gen, child);
  if (child.type === 'JSXFragment') return visitJSXFragment(gen, child);
  if (child.type === 'JSXRawExpression') return child.source;
  return 'null';
}

export default {
  name: 'kmx-react',
  fileExtensions: ['.kmx'],
  lexerRules,
  parserRules,
  generatorVisitors,
};
```
